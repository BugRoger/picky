  def analyze
    preamble

    @indexes = ["\033[1mIndexes analysis\033[m:"]
    Indexes.analyze.each_pair do |name, index|
      @indexes << "\#{\"\#{name}:\".indented_to_s}:\n\#{\"exact:\\n\#{index[:exact].indented_to_s}\".indented_to_s(4)}\n\#{\"partial*:\\n\#{index[:partial].indented_to_s}\".indented_to_s(4)}\n"
    end
    @indexes = @indexes.join "\n"
  end

  def preamble
    loc = lines_of_code File.open('app/application.rb').read

    @preamble ||= "\\033[1mApplication(s)\\033[m\nDefinition LOC:  \#{\"%4d\" % loc}\nIndexes defined: \#{\"%4d\" % Indexes.size}\n"
  end
